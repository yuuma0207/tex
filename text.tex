\def\engine{}
\ifxetex
  \def\engine{XeTeX}
\else\ifluatex
  \def\engine{LuaTeX}
\else
  \ifdefined\kanjiskip
    \def\engine{(u)pTeX}
  \else
    \def\engine{pdfTeX}
  \fi
\fi\fi
\texttt{\engine}
\endgroup
